int main() {
 using namespace Polys;
 int _n = sc.n();
 int _k = sc.n();
 int k = std::max(3, _k+1);
 Poly Delta(k);
 Delta[0] = 1;
 Delta[1] = 6;
 Delta[2] = 1;
 Delta = Delta.sqrt();
 Poly Z = Delta;
 Z[0] += Int(1);
 Z[1] += Int(1);
 Z *= Int(1) / Int(2);
 int n = (_n+1) % MOD;
 Z = Z.pow(n,n);
 Z *= Delta.inv();
 for (int i = 1; i <= _k; ++ i) {
   printf("%u ", i <= _n ? static_cast<unsigned>(Z[i]) : 0);
 }
 return 0;
}
