constexpr int N = 1005;
int n, x, p, m;
int a[N], b[N];
int stir[N][N];
void initStir() {
 stir[0][0] = 1;
 for (int i = 1; i <= m; ++i) {
   for (int j = 1; j <= i; ++j) {
     stir[i][j] = (stir[i - 1][j - 1] + 1ll * j * stir[i - 1][j]) % p; } } }
void pow_to_down() {
 for (int i = 0; i <= m; ++i) {
   for (int j = 0; j <= i; ++j) {
     b[j] = (b[j] + 1ll * stir[i][j] * a[i]) % p; } } }
int power(int a, int b) {
 int r = 1;
 for (; b; b >>= 1) {
   if (b & 1) {
     r = 1ll * r * a % p; }
   a = 1ll * a * a % p; }
 return r; }
